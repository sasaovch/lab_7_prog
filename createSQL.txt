CREATE TABLE SpaceMarine (
id SERIAL PRIMARY KEY,
name VARCHAR(50) NOT NULL,
Coordinates double precision NOT NULL,
creationDateTime timestamp NOT NULL,
health INT CHECK(health>0),
heartCount INT CHECK(heartCount > 0),
loyal BOOLEAN,
AstartesCategory astartescategory NOT NULL,
Chapter VARCHAR(50),
CHECK(heartCount <= 3),
FOREIGN KEY(Coordinates) REFERENCES coordinates(x),
FOREIGN KEY(Chapter) REFERENCES chapter(name));
